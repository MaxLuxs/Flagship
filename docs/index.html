dependencies {

import android.app.Application
import io.maxluxs.flagship.core.Flagship
import io.maxluxs.flagship.core.FlagsConfig
import io.maxluxs.flagship.core.manager.DefaultFlagsManager
import io.maxluxs.flagship.core.platform.AndroidFlagsInitializer
import io.maxluxs.flagship.provider.firebase.FirebaseProviderFactory
import kotlinx.coroutines.CoroutineScope
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.launch

class MyApp : Application() {
    override fun onCreate() {
        super.onCreate()
        
        val firebaseDefaults = mapOf(
            "new_ui" to false,
            "api_timeout" to 5000,
            "welcome_message" to "Hello"
        )
        
        val config = FlagsConfig(
            appKey = "my-app",
            environment = "production",
            providers = listOf(
                FirebaseProviderFactory.create(
                    application = this,
                    defaults = firebaseDefaults
                )
            ),
            cache = AndroidFlagsInitializer.createPersistentCache(this)
        )
        
        Flagship.configure(config)
        
        val manager = Flagship.manager() as DefaultFlagsManager
        val defaultContext = AndroidFlagsInitializer.createDefaultContext(this)
        manager.setDefaultContext(defaultContext)
        
        CoroutineScope(Dispatchers.Default).launch {
            manager.ensureBootstrap(timeoutMs = 3000)
        }
    }
}

import androidx.lifecycle.lifecycleScope
import kotlinx.coroutines.launch

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        lifecycleScope.launch {
            if (Flagship.isEnabled("new_ui")) {
                showNewUI()
            }
            
            val timeout: Int = Flagship.get("api_timeout", default = 5000)
            
            val assignment = Flagship.assign("checkout_exp")
            when (assignment?.variant) {
                "control" -> showLegacyCheckout()
                "variant_a" -> showNewCheckout()
            }
        }
    }
}

import SwiftUI

@main
struct MyApp: App {

        initializeFlagship()
    }

        WindowGroup {
            ContentView()
        }
    }
    
    private func initializeFlagship() {
        let defaults: [String: Any] = [
            "new_ui": false,
            "api_timeout": 5000,
            "welcome_message": "Hello"
        ]
        
        FlagshipSwift.shared.quickConfigure(
            appKey: "my-ios-app",
            environment: "production",
            providers: []
        )
        
        Task {
            await FlagshipSwift.shared.manager().ensureBootstrap(timeoutMs: 3000)
        }
    }
}

struct ContentView: View {
    @State private var showNewUI = false
    
    var body: some View {
        Task {
            let enabled = await FlagshipSwift.shared.manager.isEnabled(
                key: "new_ui",
                default: false
            )
            showNewUI = enabled
        }
        
        if showNewUI {
            NewUIView()
        } else {
            LegacyUIView()
        }
    }
}

Sync API (After Bootstrap)
// 1. Bootstrap first (during app initialization)
class MyApp : Application() {
    override fun onCreate() {
        super.onCreate()
        Flagship.configure(config)
        
        // Bootstrap in background
        lifecycleScope.launch {
            Flagship.manager().ensureBootstrap(timeoutMs = 3000)
        }
    }
}

// 2. Use sync methods after bootstrap (no suspend needed!)
class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        // Sync methods - instant access, no coroutines needed
        if (Flagship.isEnabledSync("new_ui")) {
            showNewUI()
        }
        
        val timeout: Int = Flagship.getSync("api_timeout", default = 5000)
        val message: String = Flagship.getSync("welcome_msg", default = "Hello")
        
        // Sync experiment assignment
        val assignment = Flagship.assignSync("checkout_exp")
        when (assignment?.variant) {
            "control" -> showLegacyCheckout()
            "variant_a" -> showNewCheckout()
            else -> showLegacyCheckout()
        }
    }
}

// Note: Sync methods throw IllegalStateException if bootstrap not completed
// Always call ensureBootstrap() first or use suspend methods
Full Manager API
// Get manager for advanced operations
val flags = Flagship.manager()

// Suspend methods with context
lifecycleScope.launch {
    val ctx = EvalContext(
        userId = "user123",
        region = "US"
    )
    
    val enabled = flags.isEnabled("new_feature", default = false, ctx = ctx)
    val value: Int = flags.value("timeout", default = 30, ctx = ctx)
    val assignment = flags.assign("exp", ctx = ctx)
}

// Sync methods (after bootstrap)
val enabledSync = flags.isEnabledSync("new_feature")
val valueSync: Int = flags.valueSync("timeout", default = 30)
val assignmentSync = flags.assignSync("exp")

// Refresh flags
flags.refresh(force = true) { providerName, success ->
    println("$providerName: $success")
}

// Overrides (debug/testing)
flags.setOverride("new_feature", FlagValue.Bool(true))
flags.clearOverride("new_feature")

// List all flags
lifecycleScope.launch {
    val allFlags = flags.listAllFlags()
    val overrides = flags.listOverrides()
}

// Bootstrap control
lifecycleScope.launch {
    val success = flags.ensureBootstrap(timeoutMs = 3000)
    if (!success) {
        log("Bootstrap timeout")
    }
}
Batch Evaluation
// Evaluate multiple flags in one call
lifecycleScope.launch {
    val flags = Flagship.evaluateFlags(
        keys = listOf("new_ui", "dark_mode", "premium_features"),
        defaults = mapOf(
            "new_ui" to FlagValue.Bool(false),
            "dark_mode" to FlagValue.Bool(true),
            "premium_features" to FlagValue.Bool(false)
        )
    )
    
    // All flags evaluated in one operation
    val newUiEnabled = flags["new_ui"]?.asBoolean() ?: false
    val darkMode = flags["dark_mode"]?.asBoolean() ?: true
}

// Batch experiment evaluation
lifecycleScope.launch {
    val experiments = Flagship.evaluateExperiments(
        keys = listOf("checkout_exp", "pricing_exp", "ui_exp"),
        ctx = EvalContext(userId = "user123")
    )
    
    // All experiments assigned in one call
    experiments["checkout_exp"]?.variant?.let { variant ->
        showCheckoutVariant(variant)
    }
}

fun onUserLogin(userId: String) {
    lifecycleScope.launch {
        Flagship.preloadForUser(
            userId = userId,
            keys = listOf(
                "personalized_ui",
                "recommendations",
                "pricing_tier"
            )
        )
    }
}
